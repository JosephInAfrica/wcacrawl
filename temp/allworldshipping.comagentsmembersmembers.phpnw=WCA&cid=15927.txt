function subValid(){
	idx = form_menu.cmbBrowse.options.selectedIndex;
	val = form_menu.cmbBrowse.options[idx].text;
	vdd = form_menu.cmbBrowse.options[idx].value;
	
	if(idx<=0)return false
	if(vdd==""){
		alert('This option is available for member only.')
		return false;
	}
	if(val=="Look-up Aid Un-Rated"){
		alert("You will be viewing the unrated Look-up Aid. To view the rated version you have to first become an AWS member.");		
	}

	if(val=="Look-up Aid Rated"){

		return true
	}


	return true;
